class A {}

class B extends A {}

class C extends B {}

public class ClassCastException2 {
  //@ ensures \result == true || \result == false;
  public static boolean f() {
    try {
      A c = new C();
      B b = (B) c;
      //@ assume true; // No ClassCastException thrown
    } catch (ClassCastException exc) {
      return false;
    }
    return true;
  }
}
