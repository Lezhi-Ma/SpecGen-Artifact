class cast1 {
  //@ requires -128 <= i && i <= 127;
  //@ ensures b == i;
  //@ ensures s == i;
  //@ ensures c == (char) i;
  //@ ensures l == i;
  //@ ensures f == i;
  //@ ensures d == i;
  //@ ensures (float) d == f;
  //@ ensures (int) d == i;
  //@ ensures i == l;
  public static void f(int i) {
    if (i < -128 || i > 127) return;
    byte b = (byte) i;
    assert b == i;
    short s = (short) i;
    assert s == i;
    char c = (char) i;
    assert c == (0xffff & i);
    long l = (long) i;
    assert l == i;
    float f = (float) i;
    assert f == i;
    double d = (double) i;
    assert d == i;
    f = (float) d;
    assert (float) d == f;
    i = (int) d;
    assert (int) d == i;
    l = (long) d;
    assert i == l;
  }
}
