import java.util.Arrays;

class StripLeadingHyphens {

    //@ requires str != null;
    //@ ensures \result != null;
    //@ ensures \result.length <= str.length;
    //@ ensures (\forall int i; 0 <= i && i < \result.length; \result[i] == str[i+2]);
    //@ ensures \result == Arrays.copyOfRange(str, 2, str.length) ==> str[0] == '-' && str[1] == '-';
    //@ ensures \result == Arrays.copyOfRange(str, 1, str.length) ==> str[0] == '-';
    static char[] stripLeadingHyphens(char[] str) {
        boolean temp_Boolean = false;
        if (str == null) {
            return null;
        }
        temp_Boolean = (str[0] == '-' && str[1] == '-');
        if (temp_Boolean) {
            return Arrays.copyOfRange(str, 2, str.length);
        } else {
            temp_Boolean = str[0] == '-';
            if (temp_Boolean) {
                return Arrays.copyOfRange(str, 1, str.length);
            }
        }
        return str;
    }
    
}
