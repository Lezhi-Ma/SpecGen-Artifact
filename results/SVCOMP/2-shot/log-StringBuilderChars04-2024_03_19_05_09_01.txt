public class StringBuilderChars04 {
  //@ requires arg != null;
  //@ ensures \result == true <==> (\forall int i; 0 <= i && i < arg.length(); arg.charAt(i) != arg.charAt(i+1));
  //@ ensures \result == false <==> (\exists int i; 0 <= i && i < arg.length(); arg.charAt(i) == arg.charAt(i+1));
  public static boolean f(String arg) {
    StringBuilder buffer = new StringBuilder(arg);

    char[] charArray = new char[buffer.length()];
    buffer.getChars(0, buffer.length(), charArray, 0);

    int i = 0;
    //@ maintaining 0 <= i && i <= charArray.length;
    //@ maintaining (\forall int k; 0 <= k && k < i; charArray[k] == buffer.charAt(k));
    //@ decreases charArray.length - i;
    for (char character : charArray) {
      System.out.print(character);
      if (character == buffer.charAt(i))
        return false;
      ++i;
    }
    return true;
  }
}
