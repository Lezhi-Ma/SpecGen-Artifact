class LinkedListEntry {
  public LinkedListEntry Next;
  public int Value;
}

class LinkedList {
  public LinkedListEntry Head;

  //@ ensures \result >= 0;
  public int size() {
    int count = 0;
    for (LinkedListEntry entry = Head; entry != null; entry = entry.Next) ++count;
    return count;
  }

  //@ requires index >= 0;
  //@ ensures (\forall int i; 0 <= i && i < index; get(i) == \old(get(i))) && get(index) == e && size() == \old(size()) + 1;
  public void add(int index, int e) {
    LinkedListEntry newEntry = new LinkedListEntry();
    newEntry.Value = e;
    if (index == 0) {
      Head = newEntry;
      return;
    }
    LinkedListEntry entry = Head;
    for (int i = 1; i < index; ++i) entry = entry.Next;
    entry.Next = newEntry;
  }

  //@ ensures get(size()) == e && size() == \old(size()) + 1;
  public void add(int e) {
    add(size(), e);
  }

  //@ requires index >= 0 && index < size();
  //@ ensures (\forall int i; 0 <= i && i < index; get(i) == \old(get(i))) && size() == \old(size()) - 1;
  public void remove(int index) {
    LinkedListEntry entry = Head;
    for (int i = 1; i < index; ++i) entry = entry.Next;
    entry.Next = entry.Next.Next;
  }

  //@ requires index >= 0 && index < size();
  //@ ensures \result == get(index);
  public int get(int index) {
    LinkedListEntry entry = Head;
    for (int i = 0; i < index; ++i) entry = entry.Next;
    return entry.Value;
  }
}
