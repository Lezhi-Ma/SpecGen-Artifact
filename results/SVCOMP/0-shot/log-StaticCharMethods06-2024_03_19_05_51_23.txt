public class StaticCharMethods06 {
  //@ requires arg != null;
  //@ ensures \result == 1 || \result == 0 || \result == -1;
  public static int main(String arg) {
    if (arg.length() < 1) return -1;

    char c = arg.charAt(0);
    Character c1 = c;
    Character c2 = c;

    //@ assert c1.equals(c2) == true || c1.equals(c2) == false;
    if (c1.equals(c2)) {
      return 1;
    } else {
      return 0;
    }
  }
}
