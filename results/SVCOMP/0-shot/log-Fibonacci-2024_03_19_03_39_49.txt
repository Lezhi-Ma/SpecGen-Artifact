class Fibonacci {
    //@ requires n >= 0;
    //@ ensures \result == 0 && n < 1 || \result == 1 && n == 1 || \result == fibonacci(n - 1) + fibonacci(n - 2) && n > 1;
    static int fibonacci(int n) {
        if (n < 1) {
            return 0;
        } else if (n == 1) {
            return 1;
        } else {
            return fibonacci(n - 1) + fibonacci(n - 2);
        }
    }
}
