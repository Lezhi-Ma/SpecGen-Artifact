class Main {

  //@ requires x >= 0.0f && x <= Long.MAX_VALUE / 2;
  //@ ensures \result == test(x);
  public static boolean f(float x) {
    if (x >= 0.0f && x <= Long.MAX_VALUE / 2) {
      Main inst = new Main();
      return inst.test(x);
    }
    return true;
  }

  //@ requires true;
  //@ ensures \result == (res > 0);
  public boolean test(float x) {

    long res = (long) ++x;
    if (res > 0) System.out.println("x >0");
    else {
      return false;
    }
    return true;
  }
}
