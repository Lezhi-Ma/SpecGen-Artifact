class Main {
  //@ ensures i == -32;
  public static void main(String[] args) {
    int i = 99;
    //@ assert i == 99;
    ++i;
    //@ assert i == 100;
    int tmp = i + 2;
    //@ assert tmp == 102;
    i = tmp;
    //@ assert i == 102;
    tmp = i - 3;
    //@ assert tmp == 99;
    i = tmp;
    //@ assert i == 99;
    i += 3;
    //@ assert i == 102;
    i -= 3;
    //@ assert i == 99;
    tmp = i * 2;
    //@ assert tmp == 198;
    i = tmp;
    //@ assert i == 198;
    tmp = i / 3;
    //@ assert tmp == 66;
    i = tmp;
    //@ assert i == 66;
    i %= 34;
    //@ assert i == 32;
    i = -i;
    //@ assert i == -32;
    assert i == -32;
    
    long l = 99;
    //@ assert l == 99L;
    ++l;
    //@ assert l == 100L;
    l += 2L;
    //@ assert l == 102L;
    l -= 3L;
    //@ assert l == 99L;
    l *= 2L;
    //@ assert l == 198L;
    l /= 3L;
    //@ assert l == 66L;
    l %= 34L;
    //@ assert l == 32L;
    l = -l;
    //@ assert l == -32L;
    assert l == -32L;
  }
}
