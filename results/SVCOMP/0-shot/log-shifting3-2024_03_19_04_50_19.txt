public class shifting3 {
  //@ requires i >= 1 && i <= 100;
  //@ ensures \result == 1 || \result == 0;
  public static int f(int i) {
    if (i < 1 || i > 100) {
      return -1;
    }

    //@ assert ((1L << i) != 1L) ==> \result == 1;
    //@ assert ((1L << i) == 1L) ==> \result == 0;

    return ((1L << i) != 1L) ? 1 : 0;
  }
}
