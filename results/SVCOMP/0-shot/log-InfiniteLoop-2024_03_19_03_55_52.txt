import java.util.Random;

public class InfiniteLoop {

  //@ requires arg != null;
  //@ ensures true;
  public static void infiniteLoop(String[] arg) {
    int i = 0;
    boolean b = new Random().nextBoolean();

    //@ loop_invariant true;
    //@ decreases 0;
    while (b) {
      i++;
    }
  }

}
