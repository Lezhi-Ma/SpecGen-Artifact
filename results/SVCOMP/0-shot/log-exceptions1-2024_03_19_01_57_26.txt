class A extends Throwable {
    //@ public invariant this instanceof A;
}

class B extends A {
    //@ public invariant this instanceof B;
}

class C extends B {
    //@ public invariant this instanceof C;
}

class D extends C {
    //@ public invariant this instanceof D;
}

class Main {
    //@ requires true;
    //@ ensures true;
    public static void main(String[] arg) {
        //@ assume true;
        try {
            D d = new D();
            C c = new C();
            B b = new B();
            A a = new A();
            A e = a;
            throw e;
        } catch (D exc) {
            //@ assert false;
        } catch (C exc) {
            //@ assert false;
        } catch (B exc) {
            //@ assert false;
        } catch (A exc) {
            //@ assert false;
        }
    }
}
