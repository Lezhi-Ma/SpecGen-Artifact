public class double2long {

  //@ requires l < 100L && l > -100L;
  //@ ensures \result == (double) l;
  public static double bar(long l) {
    assert l < 100L && l > -100L;
    return (double) l;
  }

  //@ requires true;
  //@ ensures \result == bar((long) (x < 0.0 ? x * 10.0 : x / 10.0));
  public static double foo(double x) {
    if (x < 0.0) {
      x *= 10.0;
    } else {
      x /= 10.0;
    }
    long l = (long) x;
    return bar(l);
  }

}
