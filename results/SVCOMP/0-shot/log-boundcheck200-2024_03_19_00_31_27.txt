public class boundcheck200 {

  //@ requires i >= 0;
  //@ ensures i >= 0;
  private static void recursion(int i) {
    //@ loop_invariant i >= 0;
    while (i != 0) {
      if (i > 0) {
        //@ assert i > 0;
        recursion(i - 1);
      }
      if (i < 0) {
        //@ assert false;
      }
    }
  }
}
