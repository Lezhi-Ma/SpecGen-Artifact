public class Absolute {
    //@ requires num >= Integer.MIN_VALUE && num <= Integer.MAX_VALUE;
    //@ ensures \result >= 0 && \result <= Integer.MAX_VALUE;
    //@ ensures \result == (num >= 0 ? num : -num);
    public int Absolute(int num) {
        if (0 <= num)
            return num;
        else
            return -num;
    }

    //@ requires num >= Long.MIN_VALUE && num <= Long.MAX_VALUE;
    //@ ensures \result >= 0 && \result <= Long.MAX_VALUE;
    //@ ensures \result == (num >= 0 ? num : -num);
    public long Absolute(long num) {
        if (0 <= num)
            return num;
        else
            return -num;
    }
}
