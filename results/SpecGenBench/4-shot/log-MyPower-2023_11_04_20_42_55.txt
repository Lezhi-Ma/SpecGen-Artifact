class MyPower {
    //@ requires n >= 0;
    //@ ensures \result == spec_power(x, n);
    public static int power(int x, int n) {
        int res = 1;
        //@ maintaining res == spec_power(x, i);
        //@ decreases n - i;
        for(int i = 0; i < n; i++)
            res = res * x;
        return res;
    }

    //@ requires n >= 0;
    //@ ensures \result == (int) Math.pow(x, n);
    public static int spec_power(int x, int n) {
        if (n == 0) {
            return 1;
        } else {
            return x * spec_power(x, n - 1);
        }
    }
}
