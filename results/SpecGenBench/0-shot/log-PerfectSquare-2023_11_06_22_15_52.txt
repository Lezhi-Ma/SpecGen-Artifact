class PerfectSquare {
    //@ requires num >= 0;
    //@ ensures \result == true || \result == false;
    public static boolean isPerfectSquare(int num) {
        if(num == 0)
            return true;

        //@ loop_invariant i >= 1 && i <= num;
        //@ loop_invariant (\forall int j; 1 <= j && j < i; j * j != num);
        //@ decreases num - i;
        for(int i = 1; i * i <= num; i++) {
            if(i * i == num)
                return true;
        }
        return false;
    }
}
