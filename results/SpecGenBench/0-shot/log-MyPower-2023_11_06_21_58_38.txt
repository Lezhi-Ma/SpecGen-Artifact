class MyPower {
    /**
     * Computes the power of a given number.
     * 
     * @param x the base number
     * @param n the exponent
     * @return the result of x raised to the power of n
     * @requires n >= 0
     * @ensures \result == x^n
     */
    public static int power(int x, int n) {
        int res = 1;
        for(int i = 0; i < n; i++)
            res = res * x;
        return res;
    }
}
