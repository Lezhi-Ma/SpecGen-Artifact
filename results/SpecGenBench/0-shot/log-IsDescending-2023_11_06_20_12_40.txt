public class IsDescending {
    //@ requires arr != null;
    //@ ensures \result == true || \result == false;
    public boolean isDescending (int[] arr) {
        //@ assert arr != null;
        //@ assert \fresh(arr);
        int n = arr.length;
        //@ assert n >= 0;
        if (n < 2) {
            //@ assert n < 2;
            //@ assert \result == true;
            return true;
        }
        //@ assert n >= 2;
        //@ assert \result == true || \result == false;
        for (int i = 0; i < n - 1; i++) {
            //@ assert i >= 0 && i < n - 1;
            if(arr[i] <= arr[i+1])
                //@ assert arr[i] > arr[i+1];
                //@ assert \result == false;
                return false;
        }
        //@ assert \forall int i; 0 <= i && i < n - 1; arr[i] > arr[i+1];
        //@ assert \result == true;
        return true;
    }
}
