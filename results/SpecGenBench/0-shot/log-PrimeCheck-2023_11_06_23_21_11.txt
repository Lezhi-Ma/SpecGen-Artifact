class PrimeCheck {
   
	//@ requires a > 0;
	//@ ensures \result == true || \result == false;
	public boolean isPrime(int a) {
		int i = 2;
		int mid = a/2;

		//@ loop_invariant i >= 2 && i <= mid + 1;
		//@ loop_invariant \forall int j; 2 <= j < i ==> a%j != 0;
		//@ decreases mid - i + 1;
		while (i <= mid) {
			if (a%i == 0)
				return false;
			i++;
		}
		return true;
	}
}
