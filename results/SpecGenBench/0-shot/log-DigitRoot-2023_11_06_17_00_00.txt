class DigitRoot {
    //@ requires num >= 0;
    //@ ensures \result >= 0 && \result < 10;
    public int digitRoot(int num) {
        //@ loop_invariant num >= 0;
        //@ loop_invariant (\forall int i; i >= 0 && i < num; i % 10 == 0);
        while (num >= 10) {
            int sum = 0;
            //@ loop_invariant sum >= 0;
            //@ loop_invariant (\forall int i; i >= 0 && i < num; i % 10 == 0);
            //@ decreases num;
            while (num > 0) {
                sum += num % 10;
                num /= 10;
            }
            num = sum;
        }
        return num;
    }
}
