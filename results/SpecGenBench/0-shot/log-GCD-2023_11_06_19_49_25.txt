public class GCD {
	
	//@ requires true;
	//@ ensures \result >= 1;
	public static int gcd(int num1, int num2){
		int result = 1; 
	 	num1 = (0 <= num1) ? num1 : -num1;
		num2 = (0 <= num2) ? num2 : -num2;

		//@ if (num1 == 0 && num2 == 0)
		//@ then \result == -1;
		if (num1 == 0 && num2 == 0) {
			return -1;	
		}

		//@ if (num1 == 0 || num2 == 0)
		//@ then \result == (num1 > num2) ? num1 : num2;
		if (num1 == 0 || num2 == 0) { 
			return (num1 > num2) ? num1 : num2;
		}

		//@ loop_invariant 1 <= i <= num1 && i <= num2;
		//@ loop_invariant (\forall int j; 1 <= j < i; num1 % j != 0 || num2 % j != 0);
		//@ decreases num1 - i;
		for (int i = 1; i <= num1 && i <= num2; i++) {
            if (num1 % i == 0 && num2 % i == 0) {
       			result = i;
			}
        }
		
		//@ ensures \result == result;
		return result;
	}
}
