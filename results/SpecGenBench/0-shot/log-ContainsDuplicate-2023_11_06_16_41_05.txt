import java.util.Arrays;

public class ContainsDuplicate {

    //@ requires nums != null;
    //@ ensures \result == true || \result == false;
    public boolean containsDuplicate(int[] nums) {
        Arrays.sort(nums);
        int n = nums.length;

        //@ loop_invariant 0 <= i && i <= n - 1;
        //@ loop_invariant (\forall int j; 0 <= j < i; nums[j] <= nums[j + 1]);
        //@ loop_variant n - i;
        for (int i = 0; i < n - 1; i++) {
            if (nums[i] == nums[i + 1]) {
                return true;
            }
        }
        return false;
    }
}
