class NumberOfCuts {
    //@ requires n >= 0;
    //@ ensures \result >= 0;
    public int numberOfCuts(int n) {
        //@ assert n >= 0;
        if (n == 1) {
            //@ assert n == 1;
            //@ assert \result == 0;
            //@ ensures \result == 0;
            return 0;
        }
        if (n % 2 == 0) {
            //@ assert n % 2 == 0;
            //@ assert \result == n / 2;
            //@ ensures \result == n / 2;
            return n / 2;
        }
        //@ assert n % 2 != 0;
        //@ assert \result == n;
        //@ ensures \result == n;
        return n;
    }
}
