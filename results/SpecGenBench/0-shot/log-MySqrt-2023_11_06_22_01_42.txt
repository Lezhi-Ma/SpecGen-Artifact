class MySqrt {
    //@ requires x >= 0;
    //@ ensures \result >= 0;
    public int mySqrt(int x) {
        int l = 0, r = x, ans = -1;
        //@ loop_invariant l <= r;
        //@ loop_invariant ans >= -1;
        //@ loop_invariant ans * ans <= x;
        //@ loop_invariant (ans + 1) * (ans + 1) > x;
        while (l <= r) {
            int mid = l + (r - l) / 2;
            //@ assert l <= mid && mid <= r;
            if ((long) mid * mid <= x) {
                ans = mid;
                l = mid + 1;
            } else {
                r = mid - 1;
            }
        }
        return ans;
    }
}
