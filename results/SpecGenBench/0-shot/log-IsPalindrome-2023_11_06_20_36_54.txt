class IsPalindrome {
    //@ requires s != null;
    //@ ensures \result == true || \result == false;
    public static boolean isPalindrome(String s) {
        //@ assert s != null;
        int n = s.length();

        //@ loop_invariant 0 <= i && i <= s.length();
        //@ loop_invariant (\forall int j; 0 <= j < i; s.charAt(j) == s.charAt(n - 1 - j));
        //@ decreases s.length() - i;
        for(int i = 0; i < s.length(); i++) {
            //@ assert 0 <= i && i < s.length();
            //@ assert (\forall int j; 0 <= j < i; s.charAt(j) == s.charAt(n - 1 - j));
            if(s.charAt(i) != s.charAt(n - 1 - i)) {
                //@ assert s.charAt(i) != s.charAt(n - 1 - i);
                return false;
            }
        }

        //@ assert (\forall int j; 0 <= j < s.length(); s.charAt(j) == s.charAt(n - 1 - j));
        return true;
    }
}
