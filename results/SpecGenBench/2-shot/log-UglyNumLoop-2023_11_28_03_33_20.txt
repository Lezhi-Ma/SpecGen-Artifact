class UglyNum {
    //@ requires n > 0;
    //@ ensures \result == true || \result == false;
    public static boolean isUgly(int n) {
        if (n <= 0) {
            return false;
        }
        //@ maintaining n > 0;
        //@ decreasing n;
        while (n % 2 == 0) {
            n /= 2;
        }
        //@ maintaining n > 0;
        //@ decreasing n;
        while (n % 3 == 0) {
            n /= 3;
        }
        //@ maintaining n > 0;
        //@ decreasing n;
        while (n % 5 == 0) {
            n /= 5;
        }
        return n == 1;
    }
}
