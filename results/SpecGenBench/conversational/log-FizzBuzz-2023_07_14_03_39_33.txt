class FizzBuzz {
    //@ requires n != Integer.MIN_VALUE;
    //@ ensures \result == 0 || \result == 3 || \result == 5 || \result == 8;
    public int fizzBuzz(int n) {
        int res = 0;
        if (n % 3 == 0) {
            res += 3;
        }
        if (n % 5 == 0) {
            res += 5;
        }
        return res;
    }
}
