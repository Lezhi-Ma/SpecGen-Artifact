public class Absolute {
	
	//@ requires num >= Integer.MIN_VALUE && num <= Integer.MAX_VALUE;
	//@ ensures \result >= 0;
	public /*@ pure @*/ int Absolute(int num) {
		if (0 <= num)
			return num;
		else
			return -num;
	}

	//@ requires num >= Long.MIN_VALUE && num <= Long.MAX_VALUE;
	//@ ensures \result >= 0;
	public /*@ pure @*/ long Absolute(long num) {
		if (0 <= num)
			return num;
		else
			return -num;	
	}
}
